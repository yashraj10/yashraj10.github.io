# LangGraph agent with verification

def analyze(repo, question):
  plan    = planner.understand(question)
  files   = executor.read_code(plan)
  insight = analyzer.extract(files)
  report  = summarizer.generate(insight)
  result  = verifier.validate(report)

  # citations must include file_path
  # + line_range or verifier rejects
  while not result.schema_valid:
    report = summarizer.retry(result)
    result = verifier.validate(report)

  return result

# ICU census forecasting pipeline

class ICUPredictor:
  def __init__(self):
    self.features = [
      "admission_rate_24h",
      "scheduled_discharges",
      "avg_acuity_score",
      "day_of_week",
      "seasonal_index",
    ]
    self.horizon = 24  # hours

  def predict(self, current_census):
    features = self.extract(current_census)
    forecast = self.model.predict(
      features, steps=self.horizon
    )
    return {
      "predicted_census": forecast,
      "staff_recommendation": 
        self.staffing_model(forecast)
    }
